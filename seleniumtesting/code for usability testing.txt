from selenium import webdriver
from selenium.webdriver.common.by import By
from docx import Document

# Create a new instance of the Microsoft Edge WebDriver
driver = webdriver.Edge()

# Open the registration page
driver.get("https://digitalservices.dglapm.id/register")

# Test 3: Test user flows and scenarios
try:
    # Example 1: Signing up
    # Perform necessary actions for sign-up process
    # Fill in the sign-up form
    driver.find_element(By.ID, "basic_email").send_keys("brian.tahir@gmail.com")
    driver.find_element(By.ID, "user").send_keys("Brian")
    driver.find_element(By.ID, "username").send_keys("BrianT22")    
    driver.find_element(By.ID, "basic_password").send_keys("Brian007")
    driver.find_element(By.ID, "basic_confirmpassword").send_keys("Brian007")

    # Submit the form
    driver.find_element(By.CSS_SELECTOR, "button.ant-btn-submit").click()
    # Verify if the sign-up is successful
    assert "Welcome" in driver.page_source, "Sign-up process failed"
    test_result = "Test passed: Sign-up process successful"

except AssertionError as e:
    print("User flow test failed:", str(e))
    test_result = "Test failed: " + str(e)

except Exception as e:
    print("An error occurred during user flow test:", str(e))
    test_result = "An error occurred: " + str(e)

from selenium import webdriver

# Find the target element by ID
target_element = driver.find_element(By.ID, ":0.targetLanguage")

# Get the text of the element
element_text = target_element.text

# Verify the expected text is present
expected_text = "Pilih Bahasa"
if expected_text in element_text:
    test_result = "Test passed: Element text matches expected text"
    print(test_result)
